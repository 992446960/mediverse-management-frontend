### 3.1 接口规范

- Base URL: `/api/v1`
- 认证：`Authorization: Bearer <JWT Token>`
- 响应格式：
  ```json
  {
    "code": 0,       // 0=成功，非0=错误
    "message": "ok",
    "data": { ... }
  }
  ```
- 分页参数：`?page=1&page_size=20`
- 分页响应结构：
  ```json
  {
    "total": 100,
    "page": 1,
    "page_size": 20,
    "items": [ ... ]
  }
  ```
- 错误码约定：
  - `400xx` 参数错误
  - `401xx` 未认证
  - `403xx` 无权限
  - `404xx` 资源不存在
  - `409xx` 数据冲突（如名称重复）
  - `500xx` 服务器内部错误

## 四、核心功能设计

### 4.1 认证与权限

**JWT 设计：**

```json
{
  "sub": "user_id",
  "org_id": "xxx",
  "dept_id": "xxx",
  "roles": ["user", "dept_admin"],
  "exp": 1234567890
}
```

**权限检查中间件（依赖注入）：**

```python
# 每个接口通过 Depends 注入权限校验
async def require_org_admin(current_user = Depends(get_current_user)):
    if "org_admin" not in current_user.roles and "sysadmin" not in current_user.roles:
        raise HTTPException(403)
    return current_user

# 数据隔离：所有查询自动附加 org_id 过滤（系统管理员除外）
def apply_org_filter(query, current_user):
    if "sysadmin" not in current_user.roles:
        query = query.filter(Model.org_id == current_user.org_id)
    return query
```

### 4.2 文件处理流水线

**Celery 任务链：**

```python
# 上传完成后异步触发
chain(
    parse_file.s(file_id),           # 文本提取（PDF/DOCX/XLSX 等）
    extract_knowledge.s(file_id),    # LLM 结构化抽取知识卡
    index_knowledge.s(file_id),      # 向量化 + 写入 Milvus
    update_file_status.s(file_id, "done")
).apply_async()

# 每步失败时：更新状态为 failed，写入 error_msg，记录 Task 表
# 支持 max_retries=3，指数退避：countdown = 2 ** retry_count * 30（秒）
```

**知识抽取 Prompt 设计思路：**

```
角色: 你是一个医学知识结构化专家
任务: 从以下医学文本中抽取结构化知识卡
要求:
  - 每个知识点独立成一张知识卡
  - 判断类型: 循证卡(evidence) 或 规则卡(rule)
  - 输出 JSON 数组: [{title, type, content, tags}]
  - content 使用 Markdown 格式
文本: {chunk_text}
```

**状态实时推送：**

- 使用 Redis Pub/Sub 发布状态变更事件
- 前端通过 SSE 长连接监听 `/api/v1/files/{id}/status-stream`

### 4.3 流式对话架构

```
前端                    后端                       LLM
  │                       │                          │
  │  POST /messages (SSE)  │                          │
  │──────────────────────►│                          │
  │                       │  1. 鉴权                  │
  │                       │  2. 拉取分身配置           │
  │                       │  3. 拉取授权知识库         │
  │                       │  4. 知识检索（Milvus）      │
  │                       │  5. 构建 Prompt + Context  │
  │                       │  6. 调用 LLM Stream API   │
  │                       │─────────────────────────►│
  │  data: {"type":"thinking_step"...}                │
  │◄──────────────────────│◄─ thinking_process + delta│
  │  data: {"type":"delta"...}                        │
  │◄──────────────────────│                          │
  │  data: {"type":"done"...}                         │
  │◄──────────────────────│                          │
  │                       │  7. 异步保存 Message 到 DB │
  │                       │  8. 更新 token 用量统计    │
```

### 4.5 多租户数据隔离

**核心原则：** 所有业务查询在服务层自动附加租户过滤

```python
class TenantAwareRepository:
    def __init__(self, db, current_user):
        self.db = db
        self.user = current_user

    def apply_tenant_filter(self, query, model):
        """非系统管理员强制过滤 org_id"""
        if "sysadmin" not in self.user.roles:
            query = query.filter(model.org_id == self.user.org_id)
        return query

    def check_resource_access(self, resource):
        """校验是否有权访问该资源"""
        if "sysadmin" in self.user.roles:
            return True
        if hasattr(resource, "org_id") and resource.org_id != self.user.org_id:
            raise PermissionError("无权访问该机构的资源")
        return True
```

```typescript
// stores/theme.ts
import { defineStore } from 'pinia'
import { ref, watch } from 'vue'
import { theme } from 'ant-design-vue'

export type ThemeMode = 'light' | 'dark' | 'auto'

export const useThemeStore = defineStore('theme', () => {
  // 从 localStorage 读取用户偏好，默认跟随系统
  const mode = ref<ThemeMode>(
    (localStorage.getItem('theme-mode') as ThemeMode) || 'auto'
  )

  // 实际应用的主题（解析 auto 后的结果）
  const actualTheme = ref<'light' | 'dark'>('light')

  // 监听系统主题变化
  const mediaQuery = window.matchMedia('(prefers-color-scheme: dark)')

  const updateActualTheme = () => {
    if (mode.value === 'auto') {
      actualTheme.value = mediaQuery.matches ? 'dark' : 'light'
    } else {
      actualTheme.value = mode.value
    }

    // 更新 HTML 根元素的 data-theme 属性
    document.documentElement.setAttribute('data-theme', actualTheme.value)
  }

  // 切换主题模式
  const setThemeMode = (newMode: ThemeMode) => {
    mode.value = newMode
    localStorage.setItem('theme-mode', newMode)
    updateActualTheme()
  }

  // 监听模式变化
  watch(mode, updateActualTheme, { immediate: true })

  // 监听系统主题变化
  mediaQuery.addEventListener('change', updateActualTheme)

  // 获取 Ant Design Vue 主题配置
  const getAntdTheme = () => {
    return actualTheme.value === 'dark'
      ? { algorithm: theme.darkAlgorithm }
      : { algorithm: theme.defaultAlgorithm }
  }

  return {
    mode,
    actualTheme,
    setThemeMode,
    getAntdTheme
  }
})
```

### 6.3 SSE 流式对话 Composable

```typescript
// composables/useSSE.ts
import { ref, type Ref } from 'vue'

interface Message {
  // 消息类型定义
}

interface MessageContent {
  // 消息内容定义
}

export function useSSEChat(sessionId: string) {
  const messages: Ref<Message[]> = ref([])
  const streaming = ref(false)
  let abortController: AbortController | null = null

  const sendMessage = async (content: MessageContent) => {
    streaming.value = true
    abortController = new AbortController()

    const response = await fetch(
      `/api/v1/chat/sessions/${sessionId}/messages`,
      {
        method: 'POST',
        headers: { Authorization: `Bearer ${getToken()}` },
        body: buildFormData(content),
        signal: abortController.signal
      }
    )

    const reader = response.body!.getReader()
    const decoder = new TextDecoder()
    let assistantMsg = initAssistantMessage()

    while (true) {
      const { done, value } = await reader.read()
      if (done) break

      const events = parseSSEChunks(decoder.decode(value))
      for (const event of events) {
        if (event.type === 'delta') {
          assistantMsg.parts[0].text += event.content
          messages.value = updateLastMessage(messages.value, assistantMsg)
        } else if (event.type === 'thinking_step') {
          updateThinkingStep(assistantMsg.thinking_process, event)
          messages.value = updateLastMessage(messages.value, assistantMsg)
        } else if (event.type === 'done') {
          streaming.value = false
        }
      }
    }
  }

  const stopGeneration = () => abortController?.abort()

  return { messages, streaming, sendMessage, stopGeneration }
}
```

```typescript
  }
```
```typescript
})
```

## 七、关键非功能设计

### 7.1 接口幂等性

- 文件上传接口：客户端计算文件 MD5，上传前检查是否已存在（去重）
- 会话创建：使用 `idempotency_key`（前端生成 UUID），防止重复点击创建多条记录

### 7.3 缓存策略

| 缓存对象     | Key 格式                | TTL           | 失效策略               |

| JWT 黑名单   | `jwt:blacklist:{jti}`   | 至 Token 过期 | 自动过期               |
