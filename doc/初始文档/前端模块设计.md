**状态管理（stores/auth.ts）：**

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { User, LoginParams } from '@/types/auth'
import { login, logout, refreshToken, getUserInfo } from '@/api/auth'

export const useAuthStore = defineStore('auth', () => {
  const token = ref<string>(localStorage.getItem('token') || '')
  const refreshTokenValue = ref<string>(
    localStorage.getItem('refreshToken') || ''
  )
  const user = ref<User | null>(null)

  const isAuthenticated = computed(() => !!token.value && !!user.value)

  // 登录
  const loginAction = async (params: LoginParams) => {
    const res = await login(params)
    token.value = res.data.access_token
    refreshTokenValue.value = res.data.refresh_token
    localStorage.setItem('token', res.data.access_token)
    localStorage.setItem('refreshToken', res.data.refresh_token)
    await fetchUserInfo()
  }

  // 获取用户信息
  const fetchUserInfo = async () => {
    const res = await getUserInfo()
    user.value = res.data
  }

  // 登出
  const logoutAction = async () => {
    await logout()
    token.value = ''
    refreshTokenValue.value = ''
    user.value = null
    localStorage.removeItem('token')
    localStorage.removeItem('refreshToken')
  }

  // 刷新 Token
  const refreshTokenAction = async () => {
    const res = await refreshToken(refreshTokenValue.value)
    token.value = res.data.access_token
    localStorage.setItem('token', res.data.access_token)
  }

  return {
    token,
    user,
    isAuthenticated,
    loginAction,
    logoutAction,
    fetchUserInfo,
    refreshTokenAction
  }
})
```

```vue
<template>
  <div class="organizations-page">
    <a-card>
      <!-- 搜索栏 -->
      <a-row :gutter="16" class="search-bar">
        <a-col :span="8">
          <a-input-search
            v-model:value="searchKeyword"
            :placeholder="t('common.search')"

        </a-col>

          <a-button type="primary" @click="handleAdd">
            <PlusOutlined />

          </a-button>
        </a-col>
      </a-row>

      <!-- 表格 -->
      <org-table
        :data-source="dataSource"
        :loading="loading"
        :pagination="pagination"
        @change="handleTableChange"
        @edit="handleEdit"
        @delete="handleDelete"
        @toggle-status="handleToggleStatus"
      />
    </a-card>

    <!-- 新增/编辑弹窗 -->
    <org-form
      v-model:visible="formVisible"
      :form-data="currentOrg"
      :mode="formMode"
      @success="handleFormSuccess"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, reactive, onMounted } from 'vue'
import { useI18n } from 'vue-i18n'
import { message } from 'ant-design-vue'
import { PlusOutlined } from '@ant-design/icons-vue'
import OrgTable from './components/OrgTable.vue'
import OrgForm from './components/OrgForm.vue'
import {
  getOrganizations,
  deleteOrganization,
  updateOrgStatus
} from '@/api/organizations'
import type { Organization } from './types'

const { t } = useI18n()

const searchKeyword = ref('')
const loading = ref(false)
const dataSource = ref<Organization[]>([])
const pagination = reactive({
  current: 1,
  pageSize: 20,
  total: 0
})

const formVisible = ref(false)
const formMode = ref<'add' | 'edit'>('add')
const currentOrg = ref<Organization | null>(null)

// 加载数据
const loadData = async () => {
  loading.value = true
  try {
    const res = await getOrganizations({
      page: pagination.current,
      page_size: pagination.pageSize,
      keyword: searchKeyword.value
    })
    dataSource.value = res.data.items
    pagination.total = res.data.total
  } catch (error) {
    message.error(t('common.loadFailed'))
  } finally {
    loading.value = false
  }
}

// 搜索
const handleSearch = () => {
  pagination.current = 1
  loadData()
}

// 表格变化
const handleTableChange = (pag: any) => {
  pagination.current = pag.current
  pagination.pageSize = pag.pageSize
  loadData()
}

// 新增
const handleAdd = () => {
  formMode.value = 'add'
  currentOrg.value = null
  formVisible.value = true
}

// 编辑
const handleEdit = (record: Organization) => {
  formMode.value = 'edit'
  currentOrg.value = { ...record }
  formVisible.value = true
}

// 删除
const handleDelete = async (record: Organization) => {
  try {
    await deleteOrganization(record.id)
    message.success(t('common.deleteSuccess'))
    loadData()
  } catch (error) {
    message.error(t('common.deleteFailed'))
  }
}

// 启用/禁用
const handleToggleStatus = async (record: Organization) => {
  try {
    await updateOrgStatus(record.id, {
      status: record.status === 'active' ? 'inactive' : 'active'
    })
    message.success(t('common.updateSuccess'))
    loadData()
  } catch (error) {
    message.error(t('common.updateFailed'))
  }
}

// 表单提交成功
const handleFormSuccess = () => {
  formVisible.value = false
  loadData()
}

onMounted(() => {
  loadData()
})
</script>

<style scoped>
.search-bar {
  margin-bottom: 16px;
}

.text-right {
  text-align: right;
}
</style>
```

### 5.1 文件上传组件

**文件：** `src/components/FileUploader/index.vue`

```vue
<template>
  <a-upload
    v-model:file-list="fileList"
    :multiple="multiple"
    :max-count="maxCount"
    :before-upload="beforeUpload"
    :custom-request="customRequest"
    :accept="accept"
    @change="handleChange"
  >
    <a-button>
      <UploadOutlined />
      {{ t('common.upload') }}
    </a-button>
  </a-upload>
</template>

<script setup lang="ts">
import { ref } from 'vue'
import { useI18n } from 'vue-i18n'
import { message } from 'ant-design-vue'
import { UploadOutlined } from '@ant-design/icons-vue'
import type { UploadProps } from 'ant-design-vue'
import { uploadFile } from '@/api/files'

interface Props {
  accept?: string
  maxSize?: number // MB
  maxCount?: number
  multiple?: boolean
  ownerType: 'personal' | 'dept' | 'org'
  ownerId: string
}

const props = withDefaults(defineProps<Props>(), {
  accept: '.pdf,.txt,.csv,.json,.md,.xlsx,.docx',
  maxSize: 50,
  maxCount: 20,
  multiple: true
})

const emit = defineEmits<{
  success: [fileId: string]
  error: [error: any]
}>()

const { t } = useI18n()
const fileList = ref<any[]>([])

// 上传前校验
const beforeUpload: UploadProps['beforeUpload'] = file => {
  const isLtMaxSize = file.size / 1024 / 1024 < props.maxSize
  if (!isLtMaxSize) {
    message.error(t('file.sizeExceeded', { size: props.maxSize }))
    return false
  }
  return true
}

// 自定义上传
const customRequest = async (options: any) => {
  const { file, onSuccess, onError, onProgress } = options

  try {
    const formData = new FormData()
    formData.append('file', file)
    formData.append('owner_type', props.ownerType)
    formData.append('owner_id', props.ownerId)

    const res = await uploadFile(formData, {
      onUploadProgress: (progressEvent: any) => {
        const percent = Math.round(
          (progressEvent.loaded * 100) / progressEvent.total
        )
        onProgress({ percent })
      }
    })

    onSuccess(res.data)
    emit('success', res.data.id)
    message.success(t('file.uploadSuccess'))
  } catch (error) {
    onError(error)
    emit('error', error)
    message.error(t('file.uploadFailed'))
  }
}

const handleChange: UploadProps['onChange'] = info => {
  fileList.value = info.fileList
}
</script>
```

## 七、开发规范

### 7.1 组件命名规范

- 组件文件名使用 PascalCase：`UserList.vue`
- 组件目录使用 PascalCase：`UserManagement/`
- 组件内部使用 PascalCase 注册：`<UserList />`

### 7.2 API 调用规范

```typescript
// api/users.ts
import request from './index'
import type { User, UserListParams, UserListResponse } from '@/types/user'

export const getUsers = (params: UserListParams) => {
  return request.get<UserListResponse>('/api/v1/users', { params })
}

export const getUserById = (id: string) => {
  return request.get<User>(`/api/v1/users/${id}`)
}

export const createUser = (data: Partial<User>) => {
  return request.post<User>('/api/v1/users', data)
}

export const updateUser = (id: string, data: Partial<User>) => {
  return request.put<User>(`/api/v1/users/${id}`, data)
}

export const deleteUser = (id: string) => {
  return request.delete(`/api/v1/users/${id}`)
}
```
