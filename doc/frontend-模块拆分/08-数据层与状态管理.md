## 3. 状态管理 (Zustand)

**文件**: `frontend/store/chatStore.ts`

### ChatStore (聊天会话状态)

```typescript
interface ChatSession {
  id: string
  avatarId: string
  title: string
  date: string // '今天', '昨天', '过去 7 天'
  timestamp: string
  lastMessage?: string
}

interface ChatState {
  sessions: ChatSession[]
  activeSessionId: string | null

  // Actions
  addSession: (session: ChatSession) => void
  removeSession: (id: string) => void
  renameSession: (id: string, newTitle: string) => void
  setActiveSession: (id: string | null) => void
  getSessionsByDate: () => Record<string, ChatSession[]>
}
```

### 使用方法

```typescript
import { useChatStore } from '@/store/chatStore'

function ChatComponent() {
  const { sessions, activeSessionId, addSession, removeSession } =
    useChatStore()

  // 添加新会话
  const handleNewChat = () => {
    addSession({
      id: `session-${Date.now()}`,
      avatarId: 'avatar-1',
      title: '新对话',
      date: '今天',
      timestamp: new Date().toLocaleTimeString()
    })
  }

  // 删除会话
  const handleDelete = (id: string) => {
    removeSession(id)
  }

  return (
    <div>
      {sessions.map(session => (
        <div key={session.id}>{session.title}</div>
      ))}
    </div>
  )
}
```

## 4. 工具函数

**文件**: `frontend/lib/utils.ts`

### cn() - 类名合并工具

```typescript
import { clsx, type ClassValue } from 'clsx'
import { twMerge } from 'tailwind-merge'

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}
```

**用途**: 合并 Tailwind CSS 类名，避免冲突

**使用示例**:

```typescript
cn('px-4 py-2', 'bg-blue-500', { 'text-white': true })
// 输出: "px-4 py-2 bg-blue-500 text-white"

cn('px-4', 'px-6') // 后者覆盖前者
// 输出: "px-6"
```

## 5. 自定义 Hooks

**文件**: `frontend/hooks/use-mobile.ts`

### useMobile() - 移动设备检测

```typescript
export function useMobile() {
  const [isMobile, setIsMobile] = useState(false)

  useEffect(() => {
    const checkMobile = () => {
      setIsMobile(window.innerWidth < 768)
    }

    checkMobile()
    window.addEventListener('resize', checkMobile)

    return () => window.removeEventListener('resize', checkMobile)
  }, [])

  return isMobile
}
```

**使用示例**:

```typescript
function MyComponent() {
  const isMobile = useMobile()

  return <div>{isMobile ? <MobileView /> : <DesktopView />}</div>
}
```
