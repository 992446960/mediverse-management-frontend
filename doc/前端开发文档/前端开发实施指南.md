**核心实现要点：**

1. **Token 刷新队列**：401 时尝试 refresh，刷新期间其他请求排队，刷新成功后重放，失败则清除状态跳登录
2. **响应拦截**：统一解包 `ApiResponse`，非 0 code 自动 Toast 报错
3. **请求取消**：暴露 `AbortController` 支持组件卸载时取消请求


### 4.1 通用 CRUD Composable

```typescript
// src/composables/useTableData.ts
import { ref, reactive, onMounted } from 'vue'
import type { PaginationParams, PaginatedData } from '@/types/api'

interface UseTableDataOptions<T, P extends PaginationParams> {
  fetchFn: (params: P) => Promise<PaginatedData<T>>
  defaultParams?: Partial<P>
  immediate?: boolean
}

export function useTableData<T, P extends PaginationParams = PaginationParams>(
  options: UseTableDataOptions<T, P>
) {
  const { fetchFn, defaultParams = {}, immediate = true } = options

  const data = ref<T[]>([]) as Ref<T[]>
  const loading = ref(false)
  const pagination = reactive({
    current: 1,
    pageSize: 20,
    total: 0,
  })
  const searchParams = ref<Partial<P>>({ ...defaultParams } as Partial<P>)

  const loadData = async () => {
    loading.value = true
    try {
      const params = {
        page: pagination.current,
        page_size: pagination.pageSize,
        ...searchParams.value,
      } as P
      const result = await fetchFn(params)
      data.value = result.items
      pagination.total = result.total
    } finally {
      loading.value = false
    }
  }

  const handleTableChange = (pag: { current: number; pageSize: number }) => {
    pagination.current = pag.current
    pagination.pageSize = pag.pageSize
    loadData()
  }

  const handleSearch = (params?: Partial<P>) => {
    if (params) searchParams.value = { ...searchParams.value, ...params }
    pagination.current = 1
    loadData()
  }

  const refresh = () => loadData()

  if (immediate) onMounted(loadData)

  return { data, loading, pagination, searchParams, loadData, handleTableChange, handleSearch, refresh }
}
```

```typescript
  provide(WORKSPACE_CONTEXT_KEY, context)
```

### 5.3 文件处理状态跟踪

```typescript
// src/composables/useFileProcessing.ts
import { ref, onUnmounted } from 'vue'
import { getFileStatus } from '@/api/knowledge'
import type { OwnerType, FileStatus } from '@/types/knowledge'

interface ProcessingFile {
  fileId: string
  status: FileStatus
  errorMsg: string | null
}

export function useFileProcessing(ownerType: OwnerType, ownerId: string) {
  const processingFiles = ref<Map<string, ProcessingFile>>(new Map())
  const timers = new Map<string, ReturnType<typeof setInterval>>()

  const startTracking = (fileId: string) => {
    processingFiles.value.set(fileId, { fileId, status: 'uploading', errorMsg: null })

    const timer = setInterval(async () => {
      try {
        const result = await getFileStatus(ownerType, ownerId, fileId)
        processingFiles.value.set(fileId, {
          fileId,
          status: result.status,
          errorMsg: result.error_msg,
        })
        if (result.status === 'done' || result.status === 'failed') {
          stopTracking(fileId)
        }
      } catch {
        stopTracking(fileId)
      }
    }, 5000) // 每 5 秒轮询

    timers.set(fileId, timer)
  }

  const stopTracking = (fileId: string) => {
    const timer = timers.get(fileId)
    if (timer) {
      clearInterval(timer)
      timers.delete(fileId)
    }
  }

  onUnmounted(() => {
    timers.forEach(timer => clearInterval(timer))
    timers.clear()
  })

  return { processingFiles, startTracking, stopTracking }
}
```

## 六、数字医生对话模块（Phase 4）

### 6.1 SSE 流式对话 Composable

**这是整个项目最核心的技术难点**。使用 `fetch` + `ReadableStream` 实现，而非 `EventSource`。

```typescript
// src/composables/useSSEChat.ts
import { ref, type Ref } from 'vue'

// === SSE 事件类型 ===
interface ThinkingStepEvent {
  type: 'thinking_step'
  index: number
  title: string
  status: 'processing' | 'done'
  description?: string
  duration_ms?: number
}

interface DeltaEvent {
  type: 'delta'
  content: string
}

interface DoneEvent {
  type: 'done'
  message_id: string
  tokens_used: number
}

interface ErrorEvent {
  type: 'error'
  code: number
  message: string
}

type SSEEvent = ThinkingStepEvent | DeltaEvent | DoneEvent | ErrorEvent

// === 思考步骤模型 ===
export interface ThinkingStep {
  title: string
  description: string
  status: 'processing' | 'done'
  duration_ms?: number
}

// === Composable ===
export function useSSEChat() {
  const streaming = ref(false)
  const currentText = ref('')
  const thinkingSteps = ref<ThinkingStep[]>([])
  const error = ref<string | null>(null)

  let abortController: AbortController | null = null

  const sendMessage = async (
    url: string,
    body: FormData | Record<string, unknown>,
    options?: { onDelta?: (text: string) => void; onDone?: (messageId: string) => void }
  ) => {
    streaming.value = true
    currentText.value = ''
    thinkingSteps.value = []
    error.value = null
    abortController = new AbortController()

    try {
      const token = localStorage.getItem('token')
      const isFormData = body instanceof FormData

      const response = await fetch(url, {
        method: 'POST',
        headers: {
          Authorization: `Bearer ${token}`,
          ...(isFormData ? {} : { 'Content-Type': 'application/json' }),
        },
        body: isFormData ? body : JSON.stringify(body),
        signal: abortController.signal,
      })

      if (!response.ok || !response.body) {
        throw new Error(`HTTP ${response.status}`)
      }

      const reader = response.body.getReader()
      const decoder = new TextDecoder()
      let buffer = ''

      while (true) {
        const { done, value } = await reader.read()
        if (done) break

        buffer += decoder.decode(value, { stream: true })
        const lines = buffer.split('\n')
        buffer = lines.pop() || ''

        for (const line of lines) {
          if (!line.startsWith('data: ')) continue
          const jsonStr = line.slice(6).trim()
          if (!jsonStr || jsonStr === '[DONE]') continue

          try {
            const event: SSEEvent = JSON.parse(jsonStr)
            handleSSEEvent(event, options)
          } catch {
            // 忽略解析失败的行
          }
        }
      }
    } catch (err) {
      if ((err as Error).name !== 'AbortError') {
        error.value = (err as Error).message
      }
    } finally {
      streaming.value = false
      abortController = null
    }
  }

  const handleSSEEvent = (
    event: SSEEvent,
    options?: { onDelta?: (text: string) => void; onDone?: (messageId: string) => void }
  ) => {
    switch (event.type) {
      case 'thinking_step': {
        const existing = thinkingSteps.value[event.index]
        if (existing) {
          existing.title = event.title
          existing.status = event.status
          if (event.description) existing.description = event.description
          if (event.duration_ms) existing.duration_ms = event.duration_ms
        } else {
          thinkingSteps.value[event.index] = {
            title: event.title,
            status: event.status,
            description: event.description || '',
            duration_ms: event.duration_ms,
          }
        }
        break
      }
      case 'delta':
        currentText.value += event.content
        options?.onDelta?.(currentText.value)
        break
      case 'done':
        options?.onDone?.(event.message_id)
        break
      case 'error':
        error.value = event.message
        break
    }
  }

  const stopGeneration = () => {
    abortController?.abort()
    streaming.value = false
  }

  return {
    streaming,
    currentText,
    thinkingSteps,
    error,
    sendMessage,
    stopGeneration,
  }
}
```

### 6.2 Chat Store

```typescript
// src/stores/chat.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { Session, Message } from '@/types/chat'
import * as sessionApi from '@/api/sessions'

export const useChatStore = defineStore('chat', () => {
  const sessions = ref<Session[]>([])
  const currentSessionId = ref<string | null>(null)
  const messages = ref<Message[]>([])
  const loading = ref(false)

  // 按时间分组
  const groupedSessions = computed(() => {
    const now = new Date()
    const today = new Date(now.getFullYear(), now.getMonth(), now.getDate())
    const yesterday = new Date(today.getTime() - 86400000)
    const weekAgo = new Date(today.getTime() - 7 * 86400000)

    const groups: Record<string, Session[]> = {
      today: [],
      yesterday: [],
      week: [],
      earlier: [],
    }

    for (const session of sessions.value) {
      const date = new Date(session.updated_at || session.created_at)
      if (date >= today) groups.today.push(session)
      else if (date >= yesterday) groups.yesterday.push(session)
      else if (date >= weekAgo) groups.week.push(session)
      else groups.earlier.push(session)
    }

    return groups
  })

  const currentSession = computed(() =>
    sessions.value.find(s => s.id === currentSessionId.value) ?? null
  )

  const loadSessions = async () => {
    const data = await sessionApi.getChatSessions()
    sessions.value = data.items
  }

  const createSession = async (avatarId: string) => {
    const session = await sessionApi.createChatSession({
      avatar_id: avatarId,
      idempotency_key: crypto.randomUUID(),
    })
    sessions.value.unshift(session)
    currentSessionId.value = session.id
    messages.value = []
    return session
  }

  const loadMessages = async (sessionId: string) => {
    currentSessionId.value = sessionId
    const data = await sessionApi.getMessages(sessionId)
    messages.value = data.items
  }

  const addMessage = (msg: Message) => {
    messages.value.push(msg)
  }

  const updateLastAssistantMessage = (text: string, thinkingProcess?: ThinkingStep[]) => {
    const last = messages.value[messages.value.length - 1]
    if (last && last.role === 'assistant') {
      last.parts = [{ type: 'text', text }]
      if (thinkingProcess) last.thinking_process = thinkingProcess
    }
  }

  return {
    sessions,
    currentSessionId,
    messages,
    loading,
    groupedSessions,
    currentSession,
    loadSessions,
    createSession,
    loadMessages,
    addMessage,
    updateLastAssistantMessage,
  }
})
```
